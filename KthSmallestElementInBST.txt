/**
 Time Complexity - O(N)
 Space Complexity - O(N)
 */
class Solution {
    TreeNode result = null;
    int count = 0;
    public int kthSmallest(TreeNode root, int k) {
        inorder(root, k);
        if (result == null) {
            return -1;
        }
        return result.val;
    }

    private void inorder(TreeNode root, int k) {
        if (root == null) return;

        inorder(root.left, k);
        count++;
        if (k == count) {
            result = root;
            return;
        }
        inorder(root.right, k);
    }
}
